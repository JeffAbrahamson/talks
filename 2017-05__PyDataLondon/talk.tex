\begin{frame}
  \frametitle{Cosine Similarity}
  \Huge
  \only<1-2>{
  \begin{minipage}{.4\textwidth}
  \input{dotproduct.pdf_t}
  \end{minipage}%
  \begin{minipage}{.6\textwidth}
    \only<1>{
    \begin{displaymath}
      \cos\,\theta = \frac{u\cdot v}{\parallel u\parallel \parallel v\parallel}
    \end{displaymath}
    }
    \only<2>{
    \begin{displaymath}
      \cos\,\theta = u\cdot v
    \end{displaymath}

    \bigskip
    \centerline{(if $u$ and $v$ have norm 1)}
    }
  \end{minipage}
  }
  \only<3>{
    \tt
       \gray{Le chat est orange.}\\
       {\ }[1, 1, 0, 0, 0, 1, 1] \\
       {\ }[0, 0, 1, 1, 1, 1, 0] \\
       \gray{Le chien court vite.}
  }
  \only<4>{
    \tt
    {\ }\\
    $u =$[1, 1, 0, 0, 0, 1, 1] \\
    $v =$[0, 0, 1, 1, 1, 1, 0] \\
    $u\cdot v = 0 + 0 + 0 + 0 + 0 + 1 + 0 = 1$

    \begin{displaymath}
      \cos\theta = \frac{u\cdot v}{\parallel u\parallel \parallel v\parallel}
      = \frac{1}{\sqrt{4}\cdot \sqrt{4}}
      = \frac{1}{4}
    \end{displaymath}
  }
\end{frame}

\begin{frame}
  \frametitle{Linear neuron}
  \Huge
  \begin{displaymath}
    y = b + \sum_i x_i w_i
  \end{displaymath}
  \only<2>{where
    \begin{align*}
      y & = \text{output} \\
      b & = \text{bias} \\
      x_i & = \text{$i^\mathrm{\,th}$ input} \\
      w_i & = \text{weight on $i^\mathrm{\,th}$ input}
    \end{align*}
  }
  \includegraphics<3>[width=.7\textwidth]{linear-neuron.pdf}
\end{frame}

\begin{frame}
  \frametitle{Example: handwriting recognition of digits}
  \Huge
  \begin{itemize}
  \item<1-> Input neurons: pixels
  \item<1-> Output neurons: classes (digits)
  \item<1-> Connect them all! \textit{(bipartite)}
  \item<2-> Initialize input weights to random
  \end{itemize}
